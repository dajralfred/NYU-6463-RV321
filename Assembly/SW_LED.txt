/*
 * Standalone assembly language program for NYU-6463-RV32I processor
 * The label 'reset handler' will be called upon startup.
 */
.global reset_handler
.type reset_handler,@function

reset_handler:
#Program to Read from Switch Memory @ 0x00100010
addi sp, zero, 16
lw a0, 0(sp)

#Program that Writes to LED Memory @ 0x00100014
addi sp, zero, 20
sw a0, 0(sp)

ecall
